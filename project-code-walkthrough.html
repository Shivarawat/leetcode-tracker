How a command flows

1. User runs: lct show "two sum"

2. index.js — Commander parses it:
program
  .command('show <query>')
  .action(show);  // calls show.js

3. show.js — resolves the query:
const problem = await resolveQuery(query);

4. resolveQuery.js — finds the match:
   exact ID → return immediately
   1 title match → return it
   multiple → show Inquirer picker

5. display.js — prints to terminal:
printProblem(problem);
Pattern: Every command follows the same shape — resolve query → read/write store → display output. This makes adding new commands trivial.
03 / 08 — Data Layer

store.js — JSON Persistence

src/store.js
const DATA_DIR  = path.join(os.homedir(), '.leetcode-tracker');
const DATA_FILE = path.join(DATA_DIR, 'data.json');

// Lazy init — creates dir and file on first run
const load = () => {
  if (!fs.existsSync(DATA_DIR))  fs.mkdirSync(DATA_DIR, { recursive: true });
  if (!fs.existsSync(DATA_FILE)) fs.writeFileSync(DATA_FILE, JSON.stringify({ problems: [] }));
  return JSON.parse(fs.readFileSync(DATA_FILE, 'utf-8'));
};

// Upsert: update if exists, insert if not
const upsert = (problem) => {
  const data = load();
  const idx = data.problems.findIndex((p) => String(p.id) === String(problem.id));
  if (idx >= 0) data.problems[idx] = problem; // update
  else data.problems.push(problem);            // insert
  fs.writeFileSync(DATA_FILE, JSON.stringify(data, null, 2));
};
Why ~/.leetcode-tracker? Storing in the home directory means the data persists globally regardless of which terminal or project directory you're in — same convention used by ~/.ssh, ~/.gitconfig, and ~/.npmrc. Lazy init means no setup step — the first command creates everything automatically.
04 / 08 — Smart Lookup

resolve() + resolveQuery() — ID or Name

store.js — resolve()

src/store.js
const resolve = (query) => {
  const problems = load().problems;

  // 1. Try exact ID match first
  const byId = problems.find((p) =>
    String(p.id) === String(query)
  );
  if (byId) return { problem: byId, matches: [byId] };

  // 2. Fall back to partial title match
  const matches = problems.filter((p) =>
    p.title?.toLowerCase()
      .includes(String(query).toLowerCase())
  );

  // Exactly one match → return it
  if (matches.length === 1)
    return { problem: matches[0], matches };

  // 0 or many → caller handles it
  return { problem: null, matches };
};

resolveQuery.js — with picker

src/resolveQuery.js
const resolveQuery = async (query) => {
  const { problem, matches } = resolve(query);

  // Single match — return immediately
  if (problem) return problem;

  // No matches — exit with error
  if (!matches.length) {
    console.log(chalk.red(`No match for "${query}"`));
    return null;
  }

  // Multiple matches — show picker
  const { chosen } = await inquirer.prompt([{
    type: 'list',
    name: 'chosen',
    message: `Multiple matches. Pick one:`,
    choices: matches.map((p) => ({
      name: `#${p.id} — ${p.title}`,
      value: p,
    })),
  }]);
  return chosen;
};
Why split into two functions? resolve() is pure — no side effects, easy to test. resolveQuery() adds the async Inquirer UI on top. Commands that need silent lookups (like import) use resolve() directly.

importCmd.js — Google Sheets Import

src/commands/importCmd.js
const importCmd = (filePath) => {
  const content = fs.readFileSync(filePath, 'utf-8');

  // csv-parse turns each row into a { columnName: value } object
  const records = parse(content, { columns: true, trim: true });

  records.forEach((row) => {
    const id = String(row['S. No.'] ?? row['id']).trim();
    if (!id) { skipped++; return; } // skip blank rows

    // Build approaches array from optional columns
    const approaches = [];
    const buildApproach = (suffix, tcKey, scKey) => {
      const name = (row[`Approach ${suffix}`] || '').trim();
      if (name) approaches.push({
        name,
        timeComplexity:  row[tcKey] || '',
        spaceComplexity: row[scKey] || '',
      });
    };
    buildApproach('1', 'TC', 'SC');   // required
    buildApproach('2', 'TC2', 'SC2'); // optional
    buildApproach('3', 'TC3', 'SC3'); // optional

    // Upsert — preserves existing comments/solutions if re-importing
    const existing = getById(id);
    const problem = existing || { id, solvedAt: new Date().toISOString() };
    problem.title    = row['Problem Statement'];
    problem.takeaway = row['Things to Remember'];
    upsert(problem);
  });
};
Smart re-import: If you re-import the same sheet after adding more problems, the upsert logic preserves any comments, solutions, or revisit counts you've added through the CLI — it only overwrites the sheet-level fields (title, approaches, takeaway).

// Count problems per topic
const byTopic = {};
problems.forEach((p) => {
  const t = p.topic || 'uncategorized';
  byTopic[t] = (byTopic[t] || 0) + 1;
});

// Sort by count descending
Object.entries(byTopic)
  .sort((a, b) => b[1] - a[1])
  .forEach(([topic, count]) => {
    console.log(`${topic}: ${count}`);
  });

Streak calculation

// Get unique solve dates (midnight normalized)
const dates = [...new Set(
  problems.map((p) => {
    const d = new Date(p.solvedAt);
    d.setHours(0,0,0,0);
    return d.getTime(); // timestamp
  })
)].sort((a, b) => b - a); // newest first

let streak = 0;
let current = today.getTime();

for (const date of dates) {
  // Check if date is today or yesterday
  if (date === current || date === current - 86400000) {
    streak++;
    current = date - 86400000; // step back one day
  } else break; // gap found — streak ends
}
86400000 = 24 × 60 × 60 × 1000 ms = exactly one day. Normalizing to midnight before comparing ensures two problems solved on the same day count as one streak day, not two.
08 / 08 — Interview Answers

Code-Level Interview Questions

Q: How does partial name search work without a database?

The resolve() function in store.js loads the JSON array into memory and uses Array.filter() with String.includes() for case-insensitive matching. With ~500–1000 problems this is instantaneous — no need for indexed search. If the dataset grew to tens of thousands, I'd switch to SQLite with a LIKE query or add a simple trie for prefix search.

Q: How does the import handle re-importing without losing data?

The upsert() function checks if a problem with that ID already exists. If it does, it updates only the sheet-level fields (title, approaches, takeaway) while preserving CLI-added data like comments, solutions, and revisit counts. This means you can update your Google Sheet and re-import safely without losing anything you've added through the CLI.

Q: Why does the bin field in package.json have two entries?

The bin field maps command names to the entry file. Having both "leetcode-tracker" and "lct" means npm creates symlinks for both names in the global bin directory during npm install -g. Users can type either — the short alias for daily use, the full name for clarity in scripts or documentation.

Q: What would you add next?

  • GitHub Gist sync — back up data.json to a private Gist so it persists across machines
  • SQLite migration — better querying and atomic writes once the dataset grows large
  • TUI mode — use ink (React for CLIs) for an interactive terminal dashboard
  • LeetCode API integration — auto-populate title, difficulty, and tags from the problem URL
